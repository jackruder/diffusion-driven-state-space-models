* Overview

Verifications are authored as Python source blocks, not YAML.  The config
objects below are Hydra-Zen configs imported from ~ddssm.conf.experiments.*~,
so Pyright can check imports and constructor names while the workflow functions
still produce the same resolved Hydra config artifacts as the CLI.

Each stage writes under its ~run_dir~:

- ~resolved_config.yaml~: fully resolved config for reproducibility.
- ~experiment_log.json~: stage name, config identity, overrides, resolved config
  path, run directory, and key metrics.
- stage outputs such as ~metrics.csv~, checkpoints, ~metrics.json~, PNGs, or
  variance probe summaries.

The CLI remains available for sweeps and batch jobs, but this document is the
canonical verification workflow.

* Shared Python helpers

This shared block centralizes imports used throughout the verification sections;
individual source blocks can copy only the imports they need.

from __future__ import annotations

from pathlib import Path
from typing import Any, Mapping

from ddssm.conf import (
    TrainableModulesConf,
    TrainingScalarsConf,
    TransitionDiffusionConf,
    TransitionDiffusionV2Conf,
    TransitionGaussianConf,
)
from ddssm.conf.experiments.component import (
    SyntheticDiffusionExperimentConf,
    SyntheticGaussExperimentConf,
)
from ddssm.conf.experiments.kdd import (
    KDDDiffusionExperimentConf,
    KDDGaussExperimentConf,
)
from ddssm.conf.experiments.synthetic import (
    BimodalExperimentConf,
    HarmonicExperimentConf,
    Robot2DExperimentConf,
)
from ddssm.conf.experiments.variance_probe import (
    VarianceProbeBimodalCleanExperimentConf,
    VarianceProbeBimodalNoisyExperimentConf,
    VarianceProbeLGSSMExperimentConf,
    VarianceProbeNonlinearBimodalLiftExperimentConf,
)
from ddssm.workflow import (
    ConfigGroups,
    RunMetadata,
    evaluate_config,
    train_config,
    variance_config,
    visualize_config,
)


GAUSSIAN = ConfigGroups(transition=TransitionGaussianConf)
DIFFUSION = ConfigGroups(transition=TransitionDiffusionConf)
DIFFUSION_V2 = ConfigGroups(transition=TransitionDiffusionV2Conf)


def run_train_eval_viz(
    cfg: Any,
    run_dir: str,
    *,
    groups: ConfigGroups | None = None,
    updates: Mapping[str, Any] | None = None,
    identity: str,
) -> None:
    metadata = RunMetadata(config_identity=identity)
    train_config(cfg, run_dir=run_dir, groups=groups, updates=updates, metadata=metadata)
    checkpoint = str(Path(run_dir) / "checkpoints" / "ckpt_latest.pth")
    csv_path = str(Path(run_dir) / "metrics.csv")
    evaluate_config(
        cfg,
        run_dir=run_dir,
        checkpoint_path=checkpoint,
        csv_path=csv_path,
        groups=groups,
        updates=updates,
        metadata=metadata,
    )
    visualize_config(
        cfg,
        run_dir=run_dir,
        checkpoint_path=checkpoint,
        csv_path=csv_path,
        groups=groups,
        updates=updates,
        metadata=metadata,
    )
#+END_SRC

* Variance probe workflow

#+BEGIN_SRC python
variance_presets = [
    (VarianceProbeLGSSMExperimentConf, "runs/variance_probe/lgssm"),
    (VarianceProbeBimodalCleanExperimentConf, "runs/variance_probe/bimodal_clean"),
    (VarianceProbeBimodalNoisyExperimentConf, "runs/variance_probe/bimodal_noisy"),
    (
        VarianceProbeNonlinearBimodalLiftExperimentConf,
        "runs/variance_probe/nonlinear_bimodal_lift",
    ),
]

for cfg, run_dir in variance_presets:
    metadata = RunMetadata(config_identity=f"python:variance_probe.{Path(run_dir).name}")
    train_config(cfg, run_dir=run_dir, metadata=metadata)
    variance_config(
        cfg,
        run_dir=run_dir,
        checkpoint_path=str(Path(run_dir) / "checkpoints" / "ckpt_latest.pth"),
        metadata=metadata,
    )
#+END_SRC

* Hyperparameter sweeps

Hydra multirun remains the right workflow for Optuna/Slurm sweeps because the
sweeper and launcher are Hydra plugins:

#+BEGIN_SRC sh :results output :exports both
python -m ddssm.app --multirun experiment=bimodal \
    transition=diffusion \
    'experiment.training.steps=2000' \
    'experiment.training.checkpoint_every=500' \
    'experiment.hyperparams.lambda_warmup_steps=400' \
    'experiment.hyperparams.enc_lr=tag(log, interval(1e-5, 1e-3))' \
    'experiment.hyperparams.trans_lr=tag(log, interval(1e-5, 1e-3))' \
    hydra/sweeper=ddssm_optuna \
    hydra.sweeper.n_trials=20 \
    hydra.sweeper.study_name=bimodal_lr_sweep
#+END_SRC

* Adding a new experiment

1. Add any new data-generating process to ~src/ddssm/data/synthetic.py~ or a new
   data module.
2. Define eval/viz defaults in ~src/ddssm/conf/_eval_viz.py~ when needed.
3. Add an explicit Python config object in the appropriate
   ~src/ddssm/conf/experiments/*.py~ module using ~build_experiment_conf(...)~.
4. Register it with ~store(..., group="experiment", name="my_preset")~ so the
   same config works from the CLI.
5. Add composition coverage to ~tests/test_experiment_configs.py~ and a
   verification source block here.
