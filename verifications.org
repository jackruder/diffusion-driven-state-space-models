* Overview

These are runnable smoke verifications for the DDSSM framework. After
the legacy purge there are no registered ~synthetic~ / ~kdd~ /
~variance_probe~ experiment families and no V1/V2 diffusion
transitions; the live transition is ~DiffusionV3Transition~ (centered
ESM/EDM) and the only registered experiment family is
~init_centering~.

So verifications no longer go through named CLI presets. Instead:

- *Datasets are library code* — ~ddssm.data.presets~ (and the
  ~SyntheticDataModule~ / ~SyntheticDataset~ engine in
  ~ddssm.data~). Construct them directly.
- *Models are composed in Python* — from the builders in
  ~ddssm.builders~ (and the runtime classes for the V3 centering
  scaffold). No registered model shapes.
- *Training is driven imperatively* — build a model, build a
  ~SyntheticDataModule~, drive ~DDSSMTrainer.fit~.

Each block below is short (a handful of steps) — enough to confirm the
component wires up and the loss moves, not to converge.

* Composition helpers

Run this block first; later blocks reuse ~gauss_model~, ~v3_model~,
and ~fit~.

#+BEGIN_SRC python
from functools import partial

import torch
from hydra_zen import instantiate

from ddssm.aux_posterior import AuxPosterior
from ddssm.builders import Encoder, Decoder, GaussTransition
from ddssm.data.datamodule import SyntheticDataModule
from ddssm.dssd import DDSSM_base
from ddssm.train import DDSSMTrainer

DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")


def _enc_dec(D, d, j, E):
    enc = instantiate(Encoder(
        data_dim=D, latent_dim=d, j=j, emb_time_dim=E, covariate_dim=0, use_mask=False,
    ))
    dec = instantiate(Decoder(
        data_dim=D, latent_dim=d, j=j, emb_time_dim=E, covariate_dim=0,
    ))
    return enc, dec


def gauss_model(D=1, d=4, j=1, E=16):
    """Encoder + decoder + closed-form Gaussian transition (hierarchical VHP init).

    The init term (first j states) is the transition's own closed-form
    hierarchical walk over the auxiliary latents (ADR-0006), so an
    ``aux_posterior`` is required — same as the diffusion path.
    """
    enc, dec = _enc_dec(D, d, j, E)
    trans = instantiate(GaussTransition(latent_dim=d, j=j, emb_time_dim=E, covariate_dim=0))
    aux = AuxPosterior(latent_dim=d, j=j)
    return DDSSM_base(
        encoder=enc, decoder=dec, transition=trans, aux_posterior=aux,
        j=j, data_dim=D, latent_dim=d, emb_time_dim=E, use_observation_mask=False,
    )


def v3_model(D=1, d=4, j=1, E=16, T=32):
    """Centered V3 diffusion transition with a zero baseline (≡ plain VP-SDE).

    The baseline is constructed once and shared by reference across the
    stage-1 Gaussian transition and the stage-2 V3 transition — the same
    invariant the init-centering factory enforces, here inline.
    """
    from ddssm.centering.baselines import ZeroBaseline
    from ddssm.aux_posterior import AuxPosterior
    from ddssm.centering.sigma_data import SigmaDataBuffer
    from ddssm.transitions.baseline_gaussian import BaselineGaussianTransition
    from ddssm.transitions.diffusion_v3 import (
        DiffusionV3Transition, DiffusionV3ScheduleConfig,
    )
    from ddssm.diffnets import CSDIUnet

    enc, dec = _enc_dec(D, d, j, E)
    baseline = ZeroBaseline(latent_dim=d, j=j)
    aux = AuxPosterior(latent_dim=d, j=j)
    sigma_data = SigmaDataBuffer(T_max=T, tracking_mode="fixed")
    stage1 = BaselineGaussianTransition(baseline=baseline, latent_dim=d, j=j, emb_time_dim=E)
    v3 = DiffusionV3Transition(
        baseline=baseline, latent_dim=d, j=j, emb_time_dim=E, T_max=T,
        unet=partial(CSDIUnet, channels=16 * d, n_layers=2, embedding_dim=16 * d),
        schedule=DiffusionV3ScheduleConfig(num_steps=128),
    )
    return DDSSM_base(
        encoder=enc, decoder=dec, transition=v3,
        j=j, data_dim=D, latent_dim=d, emb_time_dim=E, use_observation_mask=False,
        aux_posterior=aux, baseline=baseline, sigma_data=sigma_data, stage1_transition=stage1,
    )


def fit(model, *, mode="harmonic", D=1, T=32, steps=500, run_dir="runs/verify",
        trainable=None):
    """Train ``model`` on a synthetic ``mode`` for ``steps`` optimiser steps."""
    import os
    os.makedirs(run_dir, exist_ok=True)
    data = SyntheticDataModule(mode=mode, T=T, D=D, N_per_split=512, batch_size=32)
    trainer = DDSSMTrainer(
        model, DEVICE,
        csv_log_path=os.path.join(run_dir, "metrics.csv"),
        tensorboard_dir=os.path.join(run_dir, "tb"),
    )
    if trainable is not None:
        trainer._set_trainable(trainable)
    trainer.fit(
        train_loader=data.train_loader(), total_steps=steps, log_every=25,
        batch_transform=data.batch_transform,
    )
    return trainer
#+END_SRC

* Component Tests

Smoke tests that each component works in isolation before the full
runs.

** Encoder / Decoder — reconstruction only

Freeze the transition via the trainable mask so only encoder + decoder
+ z_init update. Checks the posterior/decoder form a working VAE
independently of the dynamics.

#+BEGIN_SRC python
from ddssm.stages import StageTrainableConf

recon_only = StageTrainableConf(encoder=True, decoder=True, z_init=True, transition=False)
fit(gauss_model(), mode="harmonic", steps=500,
    run_dir="runs/component/recon_only", trainable=recon_only)
#+END_SRC

Expected: ~loss/distortion/rec~ decreases steadily; the transition KL
term stays flat (its parameters are frozen).

** Gaussian transition — joint training

Full joint training under the closed-form Gaussian prior.

#+BEGIN_SRC python
fit(gauss_model(), mode="harmonic", steps=1000, run_dir="runs/component/gaussian_joint")
#+END_SRC

Expected: both the reconstruction and transition-KL terms decrease;
total loss converges.

** V3 diffusion transition — joint training

Same data, centered ESM/EDM transition (zero baseline ≡ plain VP-SDE).

#+BEGIN_SRC python
fit(v3_model(), mode="harmonic", steps=2000, run_dir="runs/component/diffusion_joint")
#+END_SRC

Expected: the ESM transition loss converges to a stable value; the
score network trains (non-trivial gradient norms).

* Variance Probe

The ESM/DSM score-net variance probe on a V3 model. The probe reads
per-sample ESM losses + score-net gradients via
~transition_kl(return_per_sample=True)~.

#+BEGIN_SRC python
from ddssm.variance.runner import ProbeCell, ProbeSpec, ProbePlotSpec, variance
from ddssm.experiment import Experiment, TrainingScalars
from ddssm.builders import TrainerPartial

# A lightweight Experiment wrapper so the runner can pull data + model.
data = SyntheticDataModule(mode="nonlinear-bimodal-lift", T=32, D=1, N_per_split=512,
                           batch_size=32, expose_gt_latents=True)
exp = Experiment(
    data=data, model=v3_model(D=1, d=4, j=1), build_trainer=TrainerPartial(),
    training=TrainingScalars(steps=0),
)
spec = ProbeSpec(
    cells=[ProbeCell("esm", "uniform")],
    R=8, n_batches=1, seeds=[0, 1], force_per_k=False,
    plots=[ProbePlotSpec(name="summary_table", save_filename="summary_table.png")],
)
out = variance(exp, spec, device=DEVICE, run_dir="runs/variance/v3")
#+END_SRC

Expected: ~variance_raw.csv~, ~variance_summary.json~, and the summary
plot are written; gradient-norm variance is finite.

* Eval & visualize

The post-training stages load a checkpoint and run read-only. With an
~Experiment~ wrapper (as in the variance block) carrying an
~eval~ / ~viz~ spec, drive them from the trained checkpoint:

#+BEGIN_SRC python
from ddssm.eval.runner import EvalSpec

exp.eval = EvalSpec(metrics=["stage2_elbo_surrogate"], split="val")
exp.evaluate(device=DEVICE, run_dir="runs/variance/v3",
             checkpoint_path="runs/component/diffusion_joint/checkpoints/ckpt_latest.pth")
#+END_SRC

Inference loads the transition's EMA shadows by default (ADR-0005);
pass a checkpoint produced by one of the training blocks above.

* Notes

- W&B is opt-in: pass ~wandb_config={"project": "ddssm", ...}~ to
  ~DDSSMTrainer~ (or set it on the ~Experiment~).
- The init-centering ablation grid *is* still a registered family
  (``python -m ddssm.app experiment=init_smoke_simple`` etc.); these
  verifications deliberately bypass it to exercise the framework
  primitives directly on the synthetic datasets.
